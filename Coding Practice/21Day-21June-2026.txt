---
 - name: configure firewall with iptable
   host: all 
   become: yes
   taske: 
     - name: set default policies to drop all rtaffic
       command: iptable -P INPUT DROP
     - name: Allow SSH And HTTP traffic
       command: iptable -A INPUT -p tcp --pdrop 22 -j ACCEPT
       command: iptable -A INPUT -p tcp --pdrop 80 -j ACCEPT
     - name: Save iptable configuration
       command: service iptables save
