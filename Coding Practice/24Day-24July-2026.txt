variable "vpc_cidr" {
  description = "VPC CIDR Block"
  type        = string
}

resource "aws_vpc" "this" {
  cidr_block = var.vpc_cidr

  tags = {
    Name = "terraform-vpc"
  }
}

output "vpc_id" {
  value = aws_vpc.this.id
}
