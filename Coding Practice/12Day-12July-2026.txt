---
- name: install Helm
  hosts: all
  become: yes
  tasks:
    - name: download Helm
      get_url:
        url: https://get.helm.sh/helm-v3.7.0-linux-amd64.tar.gz
        dest: /tmp/helm.tar.gz

    - name: Extract Helm
      unarchive:
        src: /tmp/helm.tar.gz
        dest: /usr/local/bin/
        remote_src: yes

    - name: Set Helm binary permissions
      file:
        path: /usr/local/bin/helm
        mode: '0755'

pipeline {
    agent any
    environment {
        IMAGE_NAME = 'java-web-app'
    }
    stages {
        stage('checkout') {
            steps {
                git 'https://github.com/ravibhadarge/sample-repo.git'
            }
        }
        stage('build docker image') {
            steps {
                script {
                    docker.build("${env.IMAGE_NAME}")
                }
            }
        }
        stage('push docker image') {
            steps {
                script {
                    docker.withRegistry('https://docker.io', 'docker-credentials') {
                        docker.image("${env.IMAGE_NAME}").push('latest')
                    }
                }
        }
        stage('deploy to kubernetes') {
           steps {
               script {
                  sh 'kubectl apply -f deployment.yaml'

                }
            }
        }
    }
}





stages:
  - build
  - push
  - deploy

build:
  stage: build
  script:
    - docker build -t ravi-web-app .

push:
  stage: push
  script:
    - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"
    - docker push ravi-web-app

deploy:
  stage: deploy
  script:
    - helm repo add stable https://charts.helm.sh/stable
