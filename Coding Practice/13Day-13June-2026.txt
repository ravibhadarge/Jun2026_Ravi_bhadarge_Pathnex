- name: Setup Nginx on multiple nodes
  hosts: all
  become: yes
  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present

    - name: Start nginx service
      service:
        name: nginx
        state: started
        enabled: yes

apiVersion: v1
kind: ConfigMap
metadata:
  name: nginx-config
data:
  nginx.conf: |
    server {
      listen 80;
      server_name localhost;
      location / {
        root   /usr/share/nginx/html;
        index  index.html index.htm;
      }
    }


---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:latest
        ports:
        - containerPort: 80
        volumeMounts:
        # Mounts the specific key as a file inside Nginx configuration directory
        - name: config-volume
          mountPath: /etc/nginx/conf.d/default.conf
          subPath: nginx.conf
      volumes:
      # Links the deployment to your ConfigMap
      - name: config-volume
        configMap:
          name: nginx-config

pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                script {
                    // Uses the standard Docker pipeline plugin syntax
                    docker.build("ravi-app:${env.BUILD_NUMBER}")
                }
            }
        }
        stage('Deploy') {
            steps {
                // kubectl commands must be executed inside a shell wrapper
                sh 'kubectl apply -f kubernetes/deployment.yaml'
            }
        }
    }
}
