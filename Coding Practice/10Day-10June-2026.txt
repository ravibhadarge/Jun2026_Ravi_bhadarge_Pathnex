            name: web-app

       - name: run the database app container

            name: database-app
            image: my-database-app
            state: started
            published_port:
             - "3306:3306"

pipeline {
   agent server001
   environment {
       IMAGE_NAME = "ravi-web-app"
   }
   stages {
      stage('build docker image') {
        steps { 
          script        
             docker.build("${env.IMAGE_NAME}")
             }
           }
        }
        stage('push the docker image')
           steps {
             scripts {
                docker.withRegistry('https://docker.io', 'docker-credentials') {
                   docker.image("${env.IMAGE_NAME}").push(latest')

                  }
               }
          }
      }
  }









stages: 
  - build:
  - push:
  - deploy

build:
  stage: build
  script: 
   - docker build -t ravi-web-app .
 
push:
  stage: push
  script: 
    - docker login -u "$CI_REGISTRY_USER"  -p "$CI_REGISTRY_PASSWOR"
    - docker push ravi-web-app
 
deploy:
  stage: deploy
  script: 
     - kubectl apply -f kubernates/deployment.yml
