---
- name: Install Nginx and Deploy Website
  hosts: webserver
  become: yes

  tasks:

    - name: Install nginx
      yum:
        name: nginx
        state: present


    - name: Start nginx service
      service:
        name: nginx
        state: started
        enabled: yes


    - name: Copy HTML file
      copy:
        src: index.html
        dest: /usr/share/nginx/html/index.html
        mode: '0644'


apiVersion: apps/v1
kind: Deployment

metadata:
  name: dev-nginx-deployment

spec:
  replicas: 2

  selector:
    matchLabels:
      app: dev-nginx

  template:
    metadata:
      labels:
        app: dev-nginx

    spec:
      containers:
        - name: nginx
          image: nginx:latest

          ports:
            - containerPort: 80


apiVersion: v1
kind: Service

metadata:
  name: dev-nginx-service

spec:

  selector:
    app: dev-nginx

  ports:
    - port: 80
      targetPort: 80

  type: LoadBalancer





#!/bin/bash

echo "CPU Usage:"
top -bn1 | grep "Cpu(s)" | awk '{print $2 + $4"%"}'


echo "Memory Usage:"
free -h | awk '/Mem:/ {print "Used: "$3" / Total: "$2}'


echo "Disk Usage:"
df -h / | awk 'NR==2 {print "Used: "$3" / Total: "$2" ("$5")"}'
 

pipeline {
    agent any

    triggers {
        cron('H 2 * * *') // every day at 2 AM
    }

    stages {

        stage('Checkout') {
            steps {
                git url: 'https://github.com/ravibhadarge/sample-java-app.git'
            }
        }

        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }

        stage('Production') {
            steps {
                echo 'Deploying to Production Environment'
            }
        }
    }
}



stages:
  - build

build:
  stage: build
  image: maven:3.8.1-jdk-17
  script:
    - git clone https://github.com/ravibhadarge/sample-java-app.git
    - cd sample-java-app
    - mvn clean package
