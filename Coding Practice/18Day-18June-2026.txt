---
  - name: Setup the ha proxy
    host: all
    become: yesy
    tasks:
       - name: install HPProxy
         yum:
           name: haproxy
           state: present
       - name: configure HAProxy
         template:
           src: haproxy
           state: present
       - name: start HAProxy service
           name: haproxy
           state: started
           enabled: yes
